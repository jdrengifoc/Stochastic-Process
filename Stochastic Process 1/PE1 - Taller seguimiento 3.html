function [G, S, zeta] = simulation(n, a)
% Votes for A.
votes_a = randsample(n, a);
zeta = 2*ones(1, n);
zeta(votes_a) = 0;
S = cumsum(zeta);
% A lead the counting in every moment?
G = all(S < 1:n);
end
% Returns all the martingale values for a given size (n) and a
% a probability for voating A (prob),
function M = martingale(n, prob)
% Martingale calculus.
zeta = [0, 2*(rand(1,n-1) > prob)];
S = fliplr(cumsum(zeta));
m = 0:(n-1);
M = S ./ (n - m);
% Plot martingale.
plot(m, M, 'LineWidth', 1.5)
grid on
line([0, n-1], [1, 1])
ylim([0, max(1.1, max(M))])
xlim([0, n-1])
ylabel('M_m'); xlabel('m')
end
